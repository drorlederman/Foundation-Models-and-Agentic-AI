% ── Custom commands ───────────────────────────────────────────────────────────

% Figure placeholder: \placeholder{width}{height}{caption}
\newcommand{\placeholder}[3]{%
  \begin{center}
    \fbox{%
      \begin{minipage}[c][#2][c]{#1}
        \centering\small\color{gray}
        [Figure: #3]
      \end{minipage}%
    }%
  \end{center}%
}

% Highlight box for a key term
\newcommand{\keyterm}[1]{%
  \begin{tcolorbox}[hitbox, title=Key Concept]
    #1
  \end{tcolorbox}%
}

% Paper citation block: \papercite{Title}{Authors}{Venue, Year}
\newcommand{\papercite}[3]{%
  \begin{tcolorbox}[hitbox, title=Paper: #1]
    \textbf{Authors:} #2\\
    \textbf{Venue:} #3
  \end{tcolorbox}%
}

% Alert / warning block
\newcommand{\warningbox}[1]{%
  \begin{tcolorbox}[alertbox, title=Warning]
    #1
  \end{tcolorbox}%
}

% Section intro frame (large centered text)
\newcommand{\sectionframe}[2]{%
  \begin{frame}[plain]
    \vfill
    \centering
    {\Large\bfseries\color{hitblue} #1}\\[0.5em]
    {\normalsize\color{gray} #2}
    \vfill
  \end{frame}%
}
